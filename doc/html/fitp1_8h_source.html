22Double_t lorentzianPeak(Double_t *x, Double_t *par) {
27 return (par[0]) / TMath::Max(1.e-10,(x[0]-par[1])*(x[0]-par[1])+ par[2]*par[2]);

30Double_t gaussianPeak(Double_t *x, Double_t *par) {
31 return par[0]*exp(-0.5* ((x[0]-par[1])/par[2]) * ((x[0]-par[1])/par[2]));
32}

36 return par[0] + par[1]*x[0];

39// Sum of background and peak function
40Double_t fitfunction(Double_t *x, Double_t *par) {
41
42 if(par[11]==0) return background(x,par) + lorentzianPeak(x,&par[2])+ lorentzianPeak(x,&par[5])+ lorentzianPeak(x,&par[8]);
43 if(par[11]==1) return background(x,par) + gaussianPeak(x,&par[2])+ gaussianPeak(x,&par[5])+ gaussianPeak(x,&par[8]);
44 else exit(1);
45}

47Double_t findPeaks(Double_t centralbin, Double_t slope, Double_t background, TH1 *h) {
48 Double_t initial_x=h->FindBin(centralbin);
49 Double_t initial_guess=(h->GetBinContent(initial_x)-background-slope*initial_x);
50 Double_t peak=initial_guess;
51 for( int i=initial_x-3; i <= initial_x+3; i++ ) {
52 if( h->GetBinContent(i) > peak ) peak = h->GetBinContent(i);
53 }
54 return peak;
55}
