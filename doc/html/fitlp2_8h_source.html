34Double_t lorentzianPeak(Double_t *x, Double_t *par) {
35 return (0.5*par[0]*par[2]/TMath::Pi()) / TMath::Max(1.e-10,(x[0]-par[1])*(x[0]-par[1])+ .25*par[2]*par[2]);
